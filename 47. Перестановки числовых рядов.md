
---
## Определение

$$ \Large
\text{Пусть } \alpha:\mathbb{N}\to\mathbb{N} \text{ — биекция множества натуральных чисел на себя;}
$$

$$ \Large
\text{перестановкой ряда } \sum_{n=1}^{\infty} a_n \text{ будем называть ряд }
$$

$$ \Large
\sum_{n=1}^{\infty} b_n,\quad \text{где } b_n = a_{\alpha(n)}.
$$

$$ \Large
\text{Если } \alpha^{-1} \text{ — обратное к } \alpha \text{ отображение, то}
$$

$$ \Large
\sum_{n=1}^{\infty} a_n \text{ является перестановкой ряда } \sum_{n=1}^{\infty} b_n,
$$

$$ \Large
\text{поскольку } a_n = b_{\alpha^{-1}(n)}.
$$
---
## Теорема

$$ \Large
\text{Пусть ряд } \sum_{n=1}^{\infty} a_n \text{ абсолютно сходится,}
$$

$$ \Large
\sum_{n=1}^{\infty} b_n \text{ — перестановка этого ряда.}
$$

$$ \Large
\Rightarrow \sum_{n=1}^{\infty} a_n = \sum_{n=1}^{\infty} b_n

$$


### Доказательство
1)
$$ \Large
\text{Предположим сначала, что } a_n \ge 0,\ \text{тогда и } b_n \ge 0\ \forall n.
$$

$$ \Large
\text{Абсолютная сходимость ряда } \sum_{n=1}^{\infty} a_n \text{ в данном случае означает}
$$

$$ \Large
\text{просто сходимость этого ряда, пусть } S = \sum_{n=1}^{\infty} a_n.
$$

$$ \Large
\text{Возьмём } N \in \mathbb{N} \text{ и введём } A(N) = \max \{\alpha(n)\ |\ 1 \le n \le N\}.
$$

$$ \Large
\text{Тогда }
$$

$$ \Large
\sum_{k=1}^{N} b_k = \sum_{n=1}^{N} a_{\alpha(n)} \le \sum_{n=1}^{A(N)} a_n \le \sum_{n=1}^{\infty} a_n = S \quad (*)

$$

$$ \Large
\text{Первое неравенство здесь следует из предположения } a_k \ge 0\ \forall k,
$$

$$ \Large
\text{и того, что среди чисел } \alpha(n),\ 1 \le n \le N \text{могут быть}
$$
$$ \Large
\text{не все числа от } 1 \text{ до } N \text{(т.к. $A(N)\ge N$)}
$$

$$ \Large
\Rightarrow
\sum_{k=1}^{\infty} b_k \le S

$$

$$ \Large
\text{Пусть } T = \sum_{k=1}^{\infty} b_k.
$$

$$ \Large
\text{Поскольку ряд } \sum_{n=1}^{\infty} a_n \text{ является перестановкой ряда } \sum_{k=1}^{\infty} b_k,
$$

$$ \Large
\text{то по уже проведённым рассуждениям с $(*)$ по аналогии) }
$$

$$ \Large
\sum_{n=1}^{\infty} a_n \le T
$$
$$ \Large
\text{получаем, что } S = T,\ \text{что и требовалось в предположении } a_n \ge 0.
$$
$$
$$
2)
$$ \Large
\text{Пусть теперь } a_n \text{ могут иметь произвольный знак. Положим}
$$

$$ \Large
a_n^{+} =
\begin{cases}
a_n, & \text{если } a_n \ge 0,\\
0, & \text{если } a_n < 0,
\end{cases}
$$

$$ \Large
a_n^{-} =
\begin{cases}
|a_n|, & \text{если } a_n \le 0,\\
0, & \text{если } a_n > 0.
\end{cases}
$$

$$ \Large
\text{Тогда } a_n = a_n^{+} - a_n^{-},\quad |a_n| = a_n^{+} + a_n^{-}.
$$

$$ \Large
\text{Поскольку ряд } \sum_{n=1}^{\infty} a_n \text{ абсолютно сходится, то}
$$

$$ \Large
S = \sum_{n=1}^{\infty} |a_n| = \sum_{n=1}^{\infty} (a_n^{+} + a_n^{-}) \in \mathbb{R}
$$

$$ \Large
 a_n^{+} \le |a_n|,\quad a_n^{-} \le |a_n|,
$$
$$ \Large
\text{то по признакам сравнения рядов с неотриц. слагаемыми следует, что сходятся}
$$

$$ \Large
\sum_{n=1}^{\infty} a_n^{+} \ \text{и}\ \sum_{n=1}^{\infty} a_n^{-}.
$$


$$ \Large
\text{Пусть } \alpha:\mathbb{N}\to\mathbb{N} \text{ — биекция, } b_n = a_{\alpha(n)},
$$

$$ \Large
\text{тогда } b_n^{+} = a_{\alpha(n)}^{+},\quad b_n^{-} = a_{\alpha(n)}^{-},
$$

$$ \Large
\text{и по первой части теоремы}
$$

$$ \Large
\sum_{n=1}^{\infty} b_n^{+} = \sum_{n=1}^{\infty} a_n^{+},\quad
\sum_{n=1}^{\infty} b_n^{-} = \sum_{n=1}^{\infty} a_n^{-}.
$$

$$ \Large
\text{Тогда получаем}
$$

$$ \Large
\sum_{n=1}^{\infty} b_n
= \sum_{n=1}^{\infty} (b_n^{+} - b_n^{-})
= \sum_{n=1}^{\infty} b_n^{+} - \sum_{n=1}^{\infty} b_n^{-} =
$$

$$ \Large
= \sum_{n=1}^{\infty} a_n^{+} - \sum_{n=1}^{\infty} a_n^{-}
= \sum_{n=1}^{\infty} a_n,
$$

$$ \Large
\text{что и требовалось.}
$$

---
$$ \Large
\text{Свойства сходящихся рядов, следующие из свойств пределов последовательностей:}
$$
1)
$$ \Large
\text{если сходятся ряды } \sum_{n=1}^{\infty} p_n \ \text{и}\ \sum_{n=1}^{\infty} q_n,
$$

$$ \Large
\text{то сходится ряд } \sum_{n=1}^{\infty} (p_n + q_n)
$$

$$ \Large
\text{и } \sum_{n=1}^{\infty} (p_n + q_n) = \sum_{n=1}^{\infty} p_n + \sum_{n=1}^{\infty} q_n;
$$
2)
$$ \Large
\text{если сходится ряд } \sum_{n=1}^{\infty} p_n \ \text{и } c \in \mathbb{R},
$$

$$ \Large
\text{то сходится ряд } \sum_{n=1}^{\infty} c p_n
$$

$$ \Large
\text{и } \sum_{n=1}^{\infty} c p_n = c \sum_{n=1}^{\infty} p_n.
$$
	---
	## Пример
	
	$$ \Large
	\text{Рассмотрим } \sum_{n=1}^{\infty} \frac{1}{n^p}
	
	$$
	
	$$ \Large
	\text{По теореме этот ряд сходится одновременно с } \int_{1}^{\infty} \frac{dx}{x^p}.
	$$
	
	$$ \Large
	\text{По примеру интеграл сходится при } p>1 \text{ и расходится при } p \le 1,
	$$
	
	$$ \Large
	\text{поэтому ряд $\sum_{n=1}^{\infty} \frac{1}{n^p}$ сходится при } p>1 \text{ и расходится при } p \le 1.
	$$
	
	$$ \Large
	\text{В частности, при } p=1 \text{ ряд } \sum_{n=1}^{\infty} \frac{1}{n} \text{ расходится.}
	$$
