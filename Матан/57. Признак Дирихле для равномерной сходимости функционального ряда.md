## Теорема
$$
\Large
\text{Пусть } \sum_{n=1}^{\infty} a_n(x)\,b_n(x),\ \ x\in X,\ \ \{b_n(x)\}_{n=1}^{\infty} \ \text{ - монотонная},\ \ b_n(x) \underset{\substack{n \to \infty \\ x \in X}}{\rightrightarrows} 0
$$

$$
\Large
\exists M:\ \forall n,\ \forall x\in X:\ |S_n(x)|\le M,\ \ \ S_n(x)=a_1(x)+\dots+a_n(x)
$$

$$
\Large
\Rightarrow \sum_{n=1}^{\infty} a_n(x)\,b_n(x) \text{ сходится на } X
$$
## Доказательство
$$
\Large
\text{Применим преобразование Абеля}
$$

$$
\Large
A_n=0
$$
$$\Large
\ A_{n+1}=a_{n+1}
$$
$$\Large
A_{n+2}=a_{n+1}+a_{n+2}\
$$
$$\Large
\dots 
$$
$$\Large
A_m=a_{n+1}+\dots+a_m
$$
$$\Large

$$
$$
\Large
a_{n+1}=A_{n+1}-A_n
$$
$$\Large a_{n+2}=A_{n+2}-A_{n+1}
$$
$$\Large
\dots
$$
$$\Large
a_m=A_m-A_{m-1}
$$
$$ \Large
\\
$$

$$
\Large
\sum_{k=n+1}^{m} a_k b_k = \sum_{k=n+1}^{m}(A_k-A_{k-1})b_k
= \sum_{k=n+1}^{m}A_k b_k - \sum_{k=n+1}^{m}A_{k-1}b_k =
$$

$$
\Large
= \sum_{k=n+1}^{m}A_k b_k - \sum_{k=n}^{m-1}A_k b_{k+1}
= A_m b_m + \sum_{k=n+1}^{m-1}A_k b_k - \sum_{k=n+1}^{m-1}A_k b_{k+1} =
$$

$$
\Large
= A_m b_m + \sum_{k=n+1}^{m-1}A_k(b_k-b_{k+1})
$$
$$ 
\text{(здесь все переходы верные - проверено)}
$$

$$
\Large
\text{Получаем } \ \sum_{k=n+1}^{m} a_k(x)b_k(x) = A_m(x)b_m(x) + \sum_{k=n+1}^{m-1} A_k(x)\big(b_k(x)-b_{k+1}(x)\big) \
$$
$$
$$
$$
$$
$$
\Large

$$

$$
\Large
k\ge n+1,\ \ A_k(x)=S_k(x)-S_n(x)
$$

$$
\Large
 |A_k(x)| = |S_k(x)-S_n(x)| \le |S_k(x)| + |S_n(x)| \le M+M=2M \
$$
$$
\Large
b_n(x) \underset{\substack{n \to \infty \\ x \in X}}{\rightrightarrows} 0 \Rightarrow \forall \varepsilon > 0\ \exists N\ \text{т.ч.}\ \forall n>N,\ \forall x\in X:\ |b_n(x)|<\varepsilon 
$$
$$
$$
$$
$$
$$
\Large
\Rightarrow\ \left|\sum_{k=n+1}^{m} a_k(x)b_k(x)\right|
\le |A_m(x)|\,|b_m(x)| + \sum_{k=n+1}^{m-1} |A_k(x)|\,|b_k(x)-b_{k+1}(x)| \le
$$

$$
\Large
\le 2M|b_m(x)| + 2M\sum_{k=n+1}^{m-1} |b_k(x)-b_{k+1}(x)| =
$$

$$
\Large
= 2M|b_m(x)| + 2M\left|\sum_{k=n+1}^{m-1} (b_k(x)-b_{k+1}(x))\right| =
$$

$$
\Large
= 2M|b_m(x)| + 2M|b_{n+1}(x)-b_m(x)| \le
$$

$$
\Large
\le 2M|b_m(x)| + 2M|b_{n+1}(x)| + 2M|b_m(x)| \le
$$

$$
\Large
\le 2M\varepsilon + 2M\varepsilon + 2M\varepsilon = 6M\varepsilon \Rightarrow
$$

$$
\Large
\Rightarrow\ \text{по критерию Коши для функционального ряда} \sum_{n=1}^{\infty} a_n(x)\,b_n(x) \text{ сходится}
