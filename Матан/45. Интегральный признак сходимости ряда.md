## Теорема

$$ \Large
\text{Пусть } f : [1,+\infty) \to \mathbb{R},\quad f(x) \ge 0,\quad f(x)\ \text{убывает}.
$$

$$ \Large
\text{Тогда ряд } \sum_{n=1}^{\infty} f(n) \text{ и несобственный интеграл } \int\limits_{1}^{\infty} f(x)\,dx
$$

$$ \Large
\text{сходятся или расходятся одновременно.}
$$

### Доказательство

$$ \Large
\text{Пусть } x \in [n,n+1],\ \text{тогда условие монотонности влечет}
$$

$$ \Large
f(n) \ge f(x) \ge f(n+1),
$$

$$ \Large
\int\limits_{n}^{n+1} f(n)\,dx \ge \int\limits_{n}^{n+1} f(x)\,dx \ge \int\limits_{n}^{n+1} f(n+1)\,dx,
$$

$$ \Large
\text{т.е. } 1\cdot f(n) \ge \int\limits_{n}^{n+1} f(x)\,dx \ge 1\cdot f(n+1).
$$

$$ \Large
\Rightarrow f(1) + \dots + f(n) \ge \int\limits_1^2 f(x)\,dx + \int\limits_2^3 f(x)\,dx + \dots + \int\limits_n^{n+1} f(x)\,dx \ge
$$

$$ \Large
\ge f(2) + \dots + f(n+1)
$$

$$ \Large
f(1) + \dots + f(n) \ge \int\limits_1^{n+1} f(x)\,dx \ge f(2) + \dots + f(n+1)

$$

$$ \Large
\text{Предположим, что } \int\limits_1^{\infty} f(x)\,dx \text{ сходится.}
$$

$$ \Large
\text{Тогда } \int\limits_1^{n+1} f(x)\,dx \le \int\limits_1^{\infty} f(x)\,dx

$$
$$ \Large
f(2)+\dots+f(n+1) \le  \int\limits_1^{\infty} f(x)\,dx

$$
$$ \Large
\text{В силу } f(x)\ge 0
$$

$$ \Large
f(1)+f(2)+\dots+f(n+1) \le f(1) + \int\limits_1^{\infty} f(x)\,dx

$$

$$ \Large
 \Rightarrow \text{ ряд } \sum_{n=1}^{\infty} f(n) \text{ сходится(его суммы ограничены и возрастают)}
$$

$$ \Large
\text{Предположим, что ряд } \sum_{n=1}^{\infty} f(n) \text{ сходится.}
$$

$$ \Large
f(1) + \dots + f(n) \ge \int\limits_1^{n+1} f(x)\,dx \ge f(2) + \dots + f(n+1) \Rightarrow

$$
$$ \Large
\Rightarrow \int\limits_1^{n+1} f(x)\,dx \le f(1)+\dots+f(n) \le \sum_{k=1}^{\infty} f(k)
$$

$$ \Large
\text{Пусть } A>1,\ \text{выберем } n \text{ так, чтобы } n+1 > A,
$$

$$ \Large
\text{тогда } 
$$

$$ \Large
\int\limits_1^A f(x)\,dx \le \int\limits_1^{n+1} f(x)\,dx \le \sum_{k=1}^{\infty} f(k)

$$

$$ \Large
\text{Следовательно } \int\limits_1^{\infty} f(x)\,dx \text{ сходится(если непонятно - см. 2.34)} 
$$

## Пример

$$ \Large
\text{Рассмотрим } \sum_{n=1}^{\infty} \frac{1}{n^p}

$$

$$ \Large
\text{По теореме этот ряд сходится одновременно с } \int_{1}^{\infty} \frac{dx}{x^p}.
$$

$$ \Large
\text{По примеру интеграл сходится при } p>1 \text{ и расходится при } p \le 1,
$$

$$ \Large
\text{поэтому ряд $\sum_{n=1}^{\infty} \frac{1}{n^p}$ сходится при } p>1 \text{ и расходится при } p \le 1.
$$

$$ \Large
\text{В частности, при } p=1 \text{ ряд } \sum_{n=1}^{\infty} \frac{1}{n} \text{ расходится.}
$$
