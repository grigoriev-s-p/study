## Еще одна формула для вычисления R
$$ \Large 
\text{Пусть } b_n \ne 0 \quad \forall n \quad \exists \lim_{n \to \infty} \frac{|b_n|}{|b_{n+1}|} = R \quad \quad \quad  0 \le R \le +\infty
$$ 
## Доказательство
$$ \Large
R_0 = \lim_{n \to \infty} \frac{|b_n|}{|b_{n+1}|}
$$
$$ \Large
0 < R_0 < +\infty
$$
1)
$$ \Large
|x-a|< R_0
$$
$$ \Large
c_n = |b_n|\cdot|x-a|^n
$$
$$ \Large
\text{Применим признак Даламбера:}
$$
$$ \Large
\frac{c_{n+1}}{c_n} = \frac{|b_{n+1}|\cdot|x-a|^{n+1}}{|b_n|\cdot|x-a|^n} = \frac{|x-a|}{\frac{|b_n|}{b_{n+1}}} \xrightarrow[n \to \infty]{} \frac{|x-a|}{R_0} < 1
$$
$$ \Large
\Rightarrow \sum_{n=1}^{\infty} c_n \ \text{ - сходится}
$$
$$ \Large
\text{т.е. для } \sum_{n=1}^{\infty} |b_n|\cdot|x-a|^n \quad x \text{ - т. сходимости при } |x-a|<R_0
$$
2)
$$ \Large
|x-a| > R_0
$$
$$ \Large
d_n = |b_n| \cdot |x-a|^n
$$
$$ \Large
\text{Снова применим признак Даламбера:}
$$
$$ \Large
\frac{d_{n+1}}{d_n} = \frac{|b_{n+1}|\cdot|x-a|^{n+1}}{|b_n|\cdot|x-a|^n} = \frac{|x-a|}{\frac{|b_n|}{b_{n+1}}} \xrightarrow[n \to \infty]{} \frac{|x-a|}{R_0} > 1
$$
$$ \Large
\Rightarrow \sum_{n=1}^{\infty} d_n \ \text{ - расходится}
$$
$$ \Large
\text{Из доказательства пр. Даламбера - $d_n$ не стремится к 0}
$$
$$ \Large
\text{т.е. } \ x_1 \ \text{ - т. расходимости } \ b_0 + \sum_{n=1}^{\infty} b_n (x-a)^n 
$$
$$
$$
$$
$$

$$ \Large
\text{Итог: }
$$
$$\Large
|x-a|<R_0 \Rightarrow x \text{ - т. сходимости}
$$
$$\Large
|x-a|>R_0 \Rightarrow x \text{ - т. расходимости}
$$
## Примеры
$$\Large
a=0
$$
1)
$$ \Large 
\sum_{n=1}^{\infty} n! x^n
$$
$$ \Large
b_n = n!
$$
$$ \Large
\frac{b_n}{b_{n+1}} = \frac{n!}{(n+1)!} = \frac{1}{n!} \xrightarrow[n \to \infty]{} 0 \quad \Rightarrow \quad R=0
$$

2)
$$\Large
\sum_{n=1}^{\infty} \frac{x^n}{n!}
$$
$$ \Large
\frac{b_n}{b_{n+1}} = \frac{1}{(n+1)!} \cdot (n+1)! = n+1 \xrightarrow[n \to \infty]{}+ \infty \quad  \Rightarrow \quad R = +\infty
$$
3)
$$\Large
\sum_{n=1}^{\infty} \frac{x^n}{n}
$$
$$\Large
\frac{b_n}{b_{n+1}} = \frac{1}{n}\cdot (n+1) = 1 + \frac{1}{n} \xrightarrow[n\to \infty]{} 1 \quad \Rightarrow \quad R=1
$$
