$$\Large  
\quad b_0 + \sum_{n=1}^{\infty} b_{n}(x-a) = S(x) \quad \quad R>0 \quad \quad I=(a-R,a+R)  
$$  
$$\Large  
\text{Пусть } \quad y \in I, y \ne a \ \Rightarrow \ \text{ наш ряд равномерно сходится на } [a,y] (\text{или на } [y,a])  
$$
$$\Large
\Rightarrow \ S(x)  \ \text{непрерывна и сходится равномерно}
$$
$$\Large
\text{Рассмотрим для $[a,y]$  (для $[y,a]$ - все аналогично, с умножением на -1)}
$$
$$\Large
\Rightarrow \int\limits_a^y S(x)dx = \int\limits_a^y b_0 dx + \sum_{n=1}^{\infty} \int\limits_a^y b_n (x-a)^n dx
$$
$$ \Large
\text{Значит } \quad \int\limits_a^y S(x)dx = b_0(y-a) + \sum_{n=1}^{\infty} \frac{b_n}{n+1} (y-a)^{n+1} = \sum_{n=1}^{\infty} \frac{b_{n-1}}{n} (y-a)^{n}
$$
---
## Теорема 
$$\Large
R_0, I_0 \text{ - радиус и интервал сходимости } \sum_{n=1}^{\infty} \frac{b_{n-1}}{n} (y-a)^{n}
$$
$$\Large
R_0 \ge R \quad \quad \quad I \subset I_0
$$
$$\Large
\Rightarrow R_0 = R \quad (I_0 = I)
$$
## Доказательство
$$ \Large
\sum_{n=1}^{\infty} \frac{b_{n}}{n+1} (y-a)^{n+1}= (y-a) \sum_{n=1}^{\infty} \frac{b_{n}}{n+1} (y-a)^{n}
$$
$$ \Large
\text{если сходится } \quad \sum_{n=1}^{\infty} \frac{b_{n}}{n+1} (y-a)^{n} \quad \text{ то сходится и }\quad \sum_{n=1}^{\infty} \frac{b_{n}}{n+1} (y-a)^{n+1}
$$ 
$$ \Large
\text{и наоборот}
$$
$$ \Large
\text{Пусть }\quad c_n = \frac{b_n}{n+1}
$$
$$\Large
\sqrt[n]{|c_n|} = \sqrt[n]{\frac{1}{n+1}} \cdot \sqrt[n]{|b_n|}
$$
$$\Large
\sqrt[n]{\frac{1}{n+1}} \xrightarrow[n \to \infty]{} 1 \text{(если не очевидно - см конец страницы)}
$$
$$\Large
\text{Рассмотрим три последовательности: } \quad \{\alpha_n\}_{n=1}^{\infty},\ \{\beta_n\}_{n=1}^{\infty},\ \{\gamma_n\}_{n=1}^{\infty}
$$ 
$$\Large
\forall n \quad \beta_n = \alpha_n \cdot \gamma_n
$$
$$\Large
\text{Примем как факт, если:} \quad \lim_{n\to \infty} \gamma_n = 1 \quad \text{то} \quad  \overline{\lim_{n \to \infty}} \alpha_n = \overline{\lim_{n \to \infty}} \beta_n
$$
$$\Large
\text{В нашем случае: } 
$$
$$\Large
\alpha_n = \sqrt[n]{|b_n|}
$$
$$\Large
\beta_n = \sqrt[n]{|c_n|}
$$
$$\Large
\gamma_n = \sqrt[n]{\frac{1}{n+1}}
$$
$$ \Large
\Rightarrow  \ \overline{\lim_{n \to \infty}}\sqrt[n]{|c_n|} = \overline{\lim_{n \to \infty}} \sqrt[n]{|b_n|}
$$
$$\Large
\Rightarrow \ R = R_0  \quad \text{чтд}
$$
