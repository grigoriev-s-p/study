$$\Large  
\quad b_0 + \sum_{n=1}^{\infty} b_{n}(x-a)^n = S(x) \quad \quad R>0 \quad \quad I=(a-R,a+R)  
$$
$$\Large
\text{ряд сходится при } x_0 = a 
$$
$$\Large
 b_0' \equiv 0 \quad \qquad (b_n(x-a)^n)'=n\cdot b_n (x-a)^{n-1}, \quad n\ge 2
$$
$$\Large
(b_1(x-a))'=b_1
$$
---
$$\Large
\text{Пусть } 0<\rho<R; \qquad \quad \text{Рассмотрим } [a-\rho,a+\rho] \subset I
$$
$$\Large
\text{Получаем 1)} \quad b_1 + \sum_{n=2}^{\infty} n b_n(x-a)^{n-1} = b_1 + \sum_{n=1}^{\infty} (n+1)b_{n+1} (x-a)^n
$$
$$\Large
\text{2)}\quad (x-a)\sum_{n=1}^{\infty} (n+1)b_{n+1} (x-a)^n = \sum_{n=1}^{\infty} (n+1)b_{n+1}(x-a)^{n+1} = \sum_{k=2}^{\infty} k b_k (x-a)^k
$$
$$\Large
\text{Если сходится } \sum_{n=1}^{\infty} (n+1)b_{n+1} (x-a)^n \text{ то сходится и } \sum_{k=2}^{\infty} k b_k (x-a)^k
$$
$$\Large
\text{и наоборот!}
$$

$$\Large
\text{Раз они сходятся одновременно - значит их $R$ равны(обозначим как $R_1$)}
$$
$$\Large
 d_k = k b_k \quad \qquad  \sqrt[k]{|d_k|} =  \sqrt[k]{k}\cdot \sqrt[k]{|b_k|}
$$
$$\Large
\sqrt[k]{k} \xrightarrow[k \to \infty]{} 1 \quad (k^{\frac{1}{k}} \xrightarrow[k \to \infty]{} 1) 
$$
$$\Large
\Rightarrow \ \overline{\lim_{k \to \infty}}\sqrt[k]{|d_k|} = \overline{\lim_{k \to \infty}} \sqrt[k]{|b_k|}
$$
$$\Large
\Rightarrow R_1 = R

$$\Large
\rho < R \Leftrightarrow \rho <R_1 
$$
$$\Large
\Rightarrow b_1 + \sum_{n=2}^{\infty} n b_n(x-a)^{n-1} \text{ равномерно сходится на } [a-\rho,a+\rho]
$$
$$\Large
\text{а также } \ b_1 + \sum_{n=2}^{\infty} n b_n(x-a)^{n-1} = S'(x), \quad x\in [a-\rho,a+\rho]
$$
$$\Large
\Rightarrow b_1 + \sum_{n=2}^{\infty} n b_n(x-a)^{n-1} = S'(x) \quad \forall x \in I (\text{в силу произвольности $\rho$})
