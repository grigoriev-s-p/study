a_1,\dots,a_n > 0,\quad b_1,\dots,b_n > 0  - \text{произвольные числа}

$$ \Large
\text{Хотим подобрать} \ t \ \text{и} \ x \ \text{для} \ a_k,\ b_k

$$ \Large

$$ \Large  
t_k x_k^p = a_k^p \quad (25), \quad 1 \le k \le n

$$ \Large  
t_k = \frac{a_k b_k}{x_k}  
$$  
  
$$ \Large  
\frac{a_k b_k}{x_k} \cdot x_k^p = a_k^p  
$$  
  
$$ \Large  
a_k b_k \cdot x_k^{p-1} = a_k^p  
$$  
  
$$ \Large  
b_k x_k^{p-1} = a_k^{p-1}  
$$  
  
$$ \Large  
x_k = \frac{a_k}{b_k^{\frac{1}{p-1}}}  
$$  
  
$$ \Large  
t_k = \frac{a_k b_k}{x_k}  
=  
\frac{a_k b_k}{a_k} \cdot b_k^{\frac{1}{p-1}}

b_k^{1 + \frac{1}{p-1}}  
=  
b_k^{\frac{p}{p-1}}  
$$  
  
$$ \Large

$$  
  
$$ \Large  
1 - \frac{1}{p} = \frac{1}{q}

a_1 b_1 + \dots + a_n b_n

\left(b_1^q + \dots + b_n^q\right)^{\frac{1}{q}}  
\left(a_1^p + \dots + a_n^p\right)^{\frac{1}{p}}
