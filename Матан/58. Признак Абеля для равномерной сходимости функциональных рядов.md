## Теорема
$$
\Large
\text{Пусть } \sum_{n=1}^{\infty} a_n(x)b_n(x),\quad \sum_{n=1}^{\infty} a_n(x)\ \text{сходится равномерно на } X
$$
$$
\Large
x\in X
$$
$$
\Large
\{b_n(x)\}_{n=1}^{\infty}\ \text{монот.}\ \forall x\in X
$$

$$
\Large
\exists K\ \text{т.ч.}\ |b_n(x)|\le K\qquad \forall n\ \text{и}\ \forall x\in X
$$
$$
\Large
\Rightarrow \sum_{n=1}^{\infty} a_n(x)b_n(x) \text{сходится равномерно на } X
$$
## Доказательство
$$
\Large
\text{Применим преобразование Абеля:}
$$

$$
\Large
A_n=0
$$
$$\Large
\ A_{n+1}=a_{n+1}
$$
$$\Large
A_{n+2}=a_{n+1}+a_{n+2}\
$$
$$\Large
\dots 
$$
$$\Large
A_m=a_{n+1}+\dots+a_m
$$
$$\Large

$$
$$
\Large
a_{n+1}=A_{n+1}-A_n
$$
$$\Large a_{n+2}=A_{n+2}-A_{n+1}
$$
$$\Large
\dots
$$
$$\Large
a_m=A_m-A_{m-1}
$$
$$ \Large
\\
$$

$$
\Large
\sum_{k=n+1}^{m} a_k b_k = \sum_{k=n+1}^{m}(A_k-A_{k-1})b_k
= \sum_{k=n+1}^{m}A_k b_k - \sum_{k=n+1}^{m}A_{k-1}b_k =
$$

$$
\Large
= \sum_{k=n+1}^{m}A_k b_k - \sum_{k=n}^{m-1}A_k b_{k+1}
= A_m b_m + \sum_{k=n+1}^{m-1}A_k b_k - \sum_{k=n+1}^{m-1}A_k b_{k+1} =
$$

$$
\Large
= A_m b_m + \sum_{k=n+1}^{m-1}A_k(b_k-b_{k+1})
$$
$$
\Large
\text{Получаем } \ \sum_{k=n+1}^{m} a_k(x)b_k(x) = A_m(x)b_m(x) + \sum_{k=n+1}^{m-1} A_k(x)\big(b_k(x)-b_{k+1}(x)\big) \
$$
$$
$$
$$
$$
$$ \Large
\text{Используем критерий Коши для функционального ряда для } \sum_{n=1}^{\infty} a_n(x)
$$
$$
\Large
\forall \varepsilon>0\ \exists N\ \text{т.ч.}\ \forall m>n>N\ \text{и}\ \forall x\in X\ \text{вып.}\ \left|a_{n+1}(x)+\dots+a_m(x)\right|<\varepsilon
$$
$$ \Large
\text{Получаем: } A_k < \varepsilon
$$
$$
\Large
\Rightarrow
\left|\sum_{k=n+1}^{m} a_k(x)b_k(x)\right|
\le
|A_m(x)|\cdot |b_m(x)|
+
\sum_{k=n+1}^{m-1}|A_k(x)|\cdot |b_k(x)-b_{k+1}(x)|
\le
$$

$$
\Large
\le
\varepsilon |b_m(x)|
+
\varepsilon \sum_{k=n+1}^{m-1}|b_k(x)-b_{k+1}(x)|
=
\varepsilon |b_m(x)|
+
\varepsilon \left|\sum_{k=n+1}^{m-1}(b_k(x)-b_{k+1}(x))\right|
=
$$

$$
\Large
=
\varepsilon |b_m(x)|
+
\varepsilon |b_{n+1}(x)-b_m(x)|
\le
\varepsilon |b_m(x)|
+
\varepsilon |b_{n+1}(x)|
+
\varepsilon |b_m(x)|
\le
3K\varepsilon
\Rightarrow
$$

$$
\Large
\Rightarrow \sum_{n=1}^{\infty} a_n(x)b_n(x) \text{ сходится равномерно на } X
$$
$$ \Large
\text{по критерию Коши для функционального ряда}
$$
